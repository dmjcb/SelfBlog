# 重载运算符

```c++
#include<iostream>

class Complex {
   public:
        Complex() = default;

        Complex(double mReal, double mImag);

        friend istream & operator >> (istream &, Complex &);
        friend ostream & operator << (ostream &, Complex &);

        Complex operator + (Complex &);
        Complex operator - (Complex &);
        Complex operator * (Complex &);
        Complex operator / (Complex &);

    private:
        double mReal;
        double mImag;
};


istream & operator >> (istream &input, Complex &c) {
    std:: >> c.mReal >> c.mImag;
}


ostream & operator << (ostream &ouput, Complex &c) {
    std:: << c.mReal << " + " << c.mImag << "i" <<std::endl;
}


Complex::Complex(double mReal, double mImag)
{
    this->mReal = mReal;
    this->mImag = mImag;
}

Complex::operator + (Complex &c) {
    return Complex(this->mReal + c.mReal, this->mImag + c.mImag);
}


Complex::operator - (Complex &c) {
    return Complex(this->mReal - c.mReal, this->mImag - c.mImag);
}


Complex::operator + (Complex &c) {
    return Complex(this->mReal + c.mReal, this->mImag + c.mImag);
}
```
