## 普通函数

    return x + y;
}
```

dll = cdll.LoadLibrary('...')

x, y = 1, 2

```

## 数组

```c
int get_array(int a[], int index) {
    int len = sizeof(a) / sizeof(a[0]);
    for (int i = 0; i < len; ++i) {
        a[i] *= 2;
    }

    return a[index];
}
```

```py
dll = cdll.LoadLibrary('...')

a = [1, 2, 3, 4, 5]

array = (c_int * 5)(*a)
res = dll.get_array(array, c_int(1))
print(res)

```


## 指针

```c++
void set_pointer(int* a) {
    *a = 0xFFFF;
}
```

```py
dll = cdll.LoadLibrary('...')

p = pointer(c_int(1))

dll.set_pointer(p)
print(p.contents)
```
