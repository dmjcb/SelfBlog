# 模板

## 函数模板

```c++
// 单参数
template<class T>
T Add(T x, T y)
{
    return x + y;
}

// 多参数
template<class T, class V>
void Print(T x, V y) {
    std::cout << x << std::endl;
    std::cout << y << std::endl;
}

// 默认参数
template<class T = uint32_t>
T Sub(T x, T y)
{
    return x - y;
}

int main(void) {
    std::cout << Add<int>(1, 2) << std::endl;

    Print<int, std::string>(4, "ABCD");

    std::cout << Sub(5, 6) << std::endl;

    return 0;
}
```

## 结构体模板

```c++
#include <iostream>
#include <vector>

template<class T>
struct Node
{
    T data;
    Node<T>* next;
};

template<class T>
void Display(Node<T>*& head, std::vector<T>& value)
{
    Node<T>* tail = head;

    for (auto v : value) {
        Node<T>* p = new Node<T>;

        p->data = v;
        p->next = nullptr;

        tail->next = p;
        tail = p;
    }


    Node<T>* p = head->next;
    while (p) {
        std::cout << p->data << " ";
        p = p->next;
    }
    std::cout << std::endl;
}


int main(void) {
    std::vector<int> value{ 1, 2, 3, 4, 5 };
    Node<int>* head = new Node<int>;

    Display<int>(head, value);

    return 0;
}
```

## 类模板

```c++
#include <iostream>

template <class T>
class Composer
{
public:
    Composer(T x, T y) {
        mX = x;
        mY = y;
    }

    T GetMax() {
        return mX > mY ? mX : mY;
    }

    T GetMin() {
        return mX <  mY ? mX : mY;
    }
    void Display() {
        std::cout << GetMax() << " " << GetMin() << std::endl;
    }

private:
    T mX;
    T mY;
};

int main(void) {
    Composer<int> com(1, 2);
    com.Display();

    return 0;
}
```
